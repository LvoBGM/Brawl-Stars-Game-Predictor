match['battleTime'] — The exact timestamp of the match.
match['event']['id'] — The unique ID of the event/map.
match['event']['mode'] — The game mode name (e.g., gemGrab, soloShowdown).
match['event']['map'] — The name of the map played.
match['battle']['mode'] — The game mode type inside the battle object.
match['battle']['type'] — The match type (e.g., ranked, friendly).
match['battle']['duration'] — Match duration in seconds (not available for Showdown).

player["name"] — The player's in-game username.
player["brawler"]["id"] — The unique game ID for the Brawler.
player["brawler"]["name"] — The name of the Brawler (e.g., SHELLY).
player["brawler"]["power"] — The Power Level of the Brawler (1 to 11).
player["brawler"]["trophies"] — The specific trophy count the player had on that Brawler.
